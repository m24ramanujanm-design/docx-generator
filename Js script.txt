function h1(text) {
  return new Paragraph({
    heading: HeadingLevel.HEADING_1,
    spacing: { before: 360, after: 120 },
    children: [new TextRun({ text, bold: true, size: 36, color: DARK_BLUE, font: "Arial" })]
  });
}
function h2(text) {
  return new Paragraph({
    heading: HeadingLevel.HEADING_2,
    spacing: { before: 280, after: 100 },
    border: { bottom: { style: BorderStyle.SINGLE, size: 4, color: MID_BLUE } },
    children: [new TextRun({ text, bold: true, size: 28, color: MID_BLUE, font: "Arial" })]
  });
}
function h3(text) {
  return new Paragraph({
    heading: HeadingLevel.HEADING_3,
    spacing: { before: 200, after: 80 },
    children: [new TextRun({ text, bold: true, size: 24, color: DARK_BLUE, font: "Arial" })]

}
function para(text, opts = {}) {
  return new Paragraph({
    spacing: { before: 80, after: 80, line: 276 },
    children: [new TextRun({ text, size: 22, font: "Arial", ...opts })]
  });
}
function bullet(text, level = 0) {
  return new Paragraph({
    numbering: { reference: "bullets", level },
    spacing: { before: 60, after: 60 },
    children: [new TextRun({ text, size: 22, font: "Arial" })]
  });
}
function numbered(text, level = 0) {
  return new Paragraph({
    numbering: { reference: "numbers", level },
    spacing: { before: 60, after: 60 },
    children: [new TextRun({ text, size: 22, font: "Arial" })]
  });
}
function spacer(before = 120) {
  return new Paragraph({ spacing: { before } });
}
function pageBreak() {
  return new Paragraph({ children: [new PageBreak()] });
}

function riskRow(category, risk, likelihood, impact, mitigation) {
  const likelihoodColor = likelihood === "High" ? "C0392B" : likelihood === "Medium" ? "E67E22" : "1A7A45";
  const impactColor = impact === "High" ? "C0392B" : impact === "Medium" ? "E67E22" : "1A7A45";
  const cellMargins = { top: 80, bottom: 80, left: 120, right: 120 };
  return new TableRow({
    children: [
      new TableCell({ borders: BORDERS, width: { size: 1440, type: WidthType.DXA }, margins: cellMargins,
        children: [new Paragraph({ children: [new TextRun({ text: category, size: 20, bold: true, font: "Arial" })] })] }),
      new TableCell({ borders: BORDERS, width: { size: 2200, type: WidthType.DXA }, margins: cellMargins,
        children: [new Paragraph({ children: [new TextRun({ text: risk, size: 20, font: "Arial" })] })] }),
      new TableCell({ borders: BORDERS, width: { size: 900, type: WidthType.DXA }, margins: cellMargins,
        shading: { fill: likelihoodColor + "22", type: ShadingType.CLEAR },
        children: [new Paragraph({ alignment: AlignmentType.CENTER, children: [new TextRun({ text: likelihood, size: 20, bold: true, color: likelihoodColor, font: "Arial" })] })] }),
      new TableCell({ borders: BORDERS, width: { size: 900, type: WidthType.DXA }, margins: cellMargins,
        shading: { fill: impactColor + "22", type: ShadingType.CLEAR },
        children: [new Paragraph({ alignment: AlignmentType.CENTER, children: [new TextRun({ text: impact, size: 20, bold: true, color: impactColor, font: "Arial" })] })] }),
      new TableCell({ borders: BORDERS, width: { size: 3120, type: WidthType.DXA }, margins: cellMargins,
        children: [new Paragraph({ children: [new TextRun({ text: mitigation, size: 20, font: "Arial" })] })] }),
    ]
  });
}

function headerRow(cols, widths) {
  const cellMargins = { top: 80, bottom: 80, left: 120, right: 120 };
  return new TableRow({
    tableHeader: true,
    children: cols.map((col, i) => new TableCell({
      borders: BORDERS,
      width: { size: widths[i], type: WidthType.DXA },
      shading: { fill: DARK_BLUE, type: ShadingType.CLEAR },
      margins: cellMargins,
      children: [new Paragraph({ alignment: AlignmentType.CENTER,
        children: [new TextRun({ text: col, bold: true, color: WHITE, size: 20, font: "Arial" })] })]
    }))
  });
}

function infoBox(title, lines, bgColor = LIGHT_BLUE, titleColor = DARK_BLUE) {
  const cellMargins = { top: 100, bottom: 100, left: 160, right: 160 };
  const children = [
    new Paragraph({ children: [new TextRun({ text: title, bold: true, size: 22, color: titleColor, font: "Arial" })] }),
    ...lines.map(l => new Paragraph({
      numbering: l.startsWith("•") ? undefined : undefined,
      spacing: { before: 40, after: 40 },
      children: [new TextRun({ text: l, size: 21, font: "Arial" })]
    }))
  ];
  return new Table({
    width: { size: 9360, type: WidthType.DXA },
    columnWidths: [9360],
    rows: [new TableRow({
      children: [new TableCell({
        borders: { top: { style: BorderStyle.SINGLE, size: 6, color: titleColor },
                   bottom: BORDER_GREY, left: { style: BorderStyle.SINGLE, size: 6, color: titleColor }, right: BORDER_GREY },
        width: { size: 9360, type: WidthType.DXA },
        shading: { fill: bgColor, type: ShadingType.CLEAR },
        margins: cellMargins,
        children
      })]
    })]
  });
}

function twoColRow(left, right, lWidth = 3000, rWidth = 6360) {
  const total = lWidth + rWidth;
  const cellMargins = { top: 80, bottom: 80, left: 120, right: 120 };
  return new TableRow({
    children: [
      new TableCell({ borders: BORDERS, width: { size: lWidth, type: WidthType.DXA }, margins: cellMargins,
        shading: { fill: GREY_BG, type: ShadingType.CLEAR },
        children: [new Paragraph({ children: [new TextRun({ text: left, bold: true, size: 21, font: "Arial" })] })] }),
      new TableCell({ borders: BORDERS, width: { size: rWidth, type: WidthType.DXA }, margins: cellMargins,
        children: [new Paragraph({ children: [new TextRun({ text: right, size: 21, font: "Arial" })] })] }),
    ]
  });
}
